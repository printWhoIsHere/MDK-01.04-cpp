#include <iostream>
#include <windows.h>
using namespace std;

char texture;

void clear();
void menu();

void paint_Line() {
    
    short length;
    char type_line;
    cout << "--------------------\n    Линия\n--------------------\n";
    cout << "[0] Выход меню\n[C] Очистить консоль\n";
    cout << "--------------------\n";
    cout << "[1] Линия\n";
    cout << "[2] Линия (Вертикальная)\n";
    cout << "[3] Слеш\n";
    cout << "[4] Слеш (Обратный)\n\n";
    cout << "--------------------\n";
    cout << "Введите размер Линии: ";
    cin >> length;
    cout << "--------------------\n";
    cout << "Выберите тип линии: ";
    cin >> type_line;
    switch (type_line)
    {
    case '0':
    {
        clear();
        menu();
    }
    case 'c':
    {
        clear();
        paint_Line();
        break;
    }
    case '1':
    {
        while (length > 0)
        {
            cout << texture << ' ';
            length -= 1;
        }
        cout << "\n"; paint_Line();
        break;
    }
    case '2':
    {
        while (length > 0)
        {
            cout << "\t" << texture << ' ' << endl;
            length -= 1;
        }
        cout << "\n"; paint_Line();
        break;
    }
    case '3':
    {
        cout << endl;
        for (int i = length; i > 0; i--)
        {
            cout << "\t";
            for (int j = 0; j <= length; j++)
            {
                if (i == j)
                {
                    cout << texture;
                }
                else
                {
                    cout << ' ';
                }
            }
            cout << endl;
        }
        cout << "/n"; paint_Line();
        break;
    }
    case '4':
    {
        for (int i = 0; i < length; i++)
        {
            cout << "\t";
            for (int j = 0; j <= length; j++)
            {
                if (i == j)
                {
                    cout << texture;
                }
                else
                {
                    cout << ' ';
                }
            }
            cout << endl;
        }
        cout << "/n"; paint_Line();
        break;
    }
    default: { clear(); paint_Line(); }
    }
}

void paint_Square() {
    cout << "--------------------\n    Квадрат\n--------------------\n";
    cout << "Введите длинну стороны: ";
    short shape_size;
    cin >> shape_size;

    {
        cout << "\t";
        for (int j = 0; j < shape_size; j++)
        {
            cout << texture << ' ';
        }
        cout << endl;
    }
    menu();
}
void paint_Rectangle() {
    short width, height;
    cout << "--------------------\n    Прямоугольник\n--------------------\n";
    cout << "Введите длину Прямоугольника: ";
    cin >> height;
    cout << "Введите ширину Прямоугольника: ";
    cin >> width;
    cout << "--------------------\n";

    short type_rectangle;
    cout << "\nВыберите тип прямоугольника: ";
    cout << "\n--------------------\n";
    cout << "\n[1] Заполненный Прямоугольник: ";
    cout << "\n[2] Не заполненный Прямоугольник: ";
    cout << "\n--------------------\nТип: ";
    cin >> type_rectangle;

    switch (type_rectangle) {
    case 1: {
        for (int i = 0; i < width; i++) {
            cout << "\t";
            for (int j = 0; j < height; j++) {
                cout << texture << ' ';
            }
            cout << endl;
        }
        menu(); break;
    }
    case 2: {
        for (int i = 0; i < width; i++) {
            cout << " ";
            for (int j = 0; j < height; j++) {
                if ((i == 0 or i == width - 1) or (j == 0 or j == height - 1))
                    cout << texture;
                else
                    cout << " ";
            }
            cout << endl; paint_Rectangle();
        }
        menu(); break;
    }
    default: {
        clear();
        paint_Rectangle();
        break;
    }
    }
}
void paint_triangle() {

    cout << "--------------------\n    Треугольник\n--------------------\n";
    cout << "Введите высоту Треугольника: ";
    short height;
    cin >> height;
    int a = 0;
    for (int i = height; i > 0; i--)
    {
        cout << "\t";
        for (int j = 0; j < i; j++)
        {
            cout << ' ';
        }
        for (int i = 0; i <= a * 2; i++)
        {
            cout << texture;
        }
        cout << endl;
        a++;
    }
    menu();
}
void paint_Cicl() {
    cout << "--------------------\n    Круг\n--------------------\n";
    cout << "Введите радиус: ";
    int radius;
    cin >> radius;
    for (int y = radius; y > 0 - radius; y--)
    {
        for (int x = -radius; x <= radius; x++)
        {
            if (((x * x) + (y * y)) <= radius * radius)
                cout << texture;
            else
                cout << " ";
        }
        cout << "\n";
    }
    menu();
}
void paint_plus() {
    cout << "--------------------\n    Плюс\n--------------------\n";
    cout << "Введите длинну Плюсы: ";
    short plus_size;
    cin >> plus_size;
    cout << "--------------------\n";
    for (int i = 0; i < plus_size; i++)
    {
        for (int j = 0; j < plus_size; j++)
        {
            if (j == plus_size / 2 || i == plus_size / 2)
            cout << texture << ' ';
            else
            cout << "  ";
        }
        cout << "\n";
    }
    menu();
}

void paint_Square_in_Square() {
    cout << "--------------------\n    Квадрат\n--------------------\n";
    cout << "Введите длинну стороны: ";
    short shape_size;
    cin >> shape_size;
    cout << "--------------------\n";
    for (int i = 0; i < shape_size; i++)
    {
        for (int j = 0; j < shape_size; j++) {
            if ((min(min(i, shape_size - 1 - i), min(j, shape_size - 1 - j)) % 2) == 0)
                cout << texture << ' ';
            else
                cout << ' ';
        }

        cout << endl;

    }
    menu();
}
