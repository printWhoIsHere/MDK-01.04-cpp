#include <iostream>
using namespace std;

int main()
{
    setlocale(0, "");
    cout << "\a\t\t***Приветствуем Вас в программе — \"Геометрические фигуры\"***\t\t\t\n";

    int figure, shape_size;
    cout << "\n\aВыберите геометрическую фигуру:" << endl;
    cout << "[1] Линия\n";
    cout << "[2] Квадрат\n";
    cout << "[3] Прямоугольник\n";
    cout << "[4] Треугольник\n";
    cout << "[5] Круг\n";
    cout << "[0] Выход\n\n";

    cout << "Геометрическая фигура №: ";
    cin >> figure;

    cout << "Выберите текстуру для вашей фигуры: ";
    char texture;
    cin >> texture;

    switch (figure)
    {
    case 1:
    {
        int type_line;
        cout << "\n\aВыберите тип Линии:" << endl;
        cout << "[1] Линия\n";
        cout << "[2] Линия (Вертикальная)\n";
        cout << "[3] Слеш\n";
        cout << "[4] Слеш (Обратный)\n\n";

        cout << "Линия №: ";
        cin >> type_line;

        switch (type_line)
        {
        case 1:
        {
            cout << "Введите размер Линии: ";
            cin >> shape_size;
            cout << endl << endl << "\t";

            while (shape_size > 0)
            {
                cout << texture << ' ';
                shape_size -= 1;
            }
            break;
        }
        case 2:
        {
            cout << "Введите размер Линии (Вертикальной): ";
            cin >> shape_size;
            cout << endl << endl;

            while (shape_size > 0)
            {
                cout << "\t" << texture << ' ' << endl;
                shape_size -= 1;
            }
            break;
        }
        case 3:
        {
            cout << "Введите размер Слеша: ";
            cin >> shape_size;
            cout << endl;

            for (int i = shape_size; i > 0; i--)
            {
                cout << "\t";
                for (int j = 0; j <= shape_size; j++)
                {
                    if (i == j)
                    {
                        cout << texture;
                    }
                    else
                    {
                        cout << ' ';
                    }
                }
                cout << endl;
            }
            cout << endl;
            break;
        }
        case 4:
        {
            cout << "Введите размер Слеша (Обратного): ";
            cin >> shape_size;
            cout << endl;

            for (int i = 0; i < shape_size; i++)
            {
                cout << "\t";
                for (int j = 0; j <= shape_size; j++)
                {
                    if (i == j)
                    {
                        cout << texture;
                    }
                    else
                    {
                        cout << ' ';
                    }
                }
                cout << endl;
            }
            break;
        }
        default: { cout << endl << "В данном списке такой линии не имеется"; break; }
        }
        break;
    }
    case 2:
    {
        cout << "Введите размер Квадрата: ";
        cin >> shape_size;
        cout << endl << endl;

        for (int i = 0; i < shape_size; i++)
        {
            cout << "\t";
            for (int j = 0; j < shape_size; j++)
            {
                cout << texture << ' ';
            }
            cout << endl;
        }
        break;
    }
    case 3:
    {
        int width, height;
        cout << "Введите длину Прямоугольника: ";
        cin >> height;
        cout << "Введите ширину Прямоугольника: ";
        cin >> width;

        int type_rectangle;
        cout << "\nВведите длину Прямоугольника: ";
        cout << "\n[1] Сплошной Прямоугольник: ";
        cout << "\n[2] Полый Прямоугольник: ";
        cin >> type_rectangle;
        cout << endl << endl;

        switch (type_rectangle)
        {
        case 1:
        {
            for (int i = 0; i < width; i++)
            {
                cout << "\t";
                for (int j = 0; j < height; j++)
                {
                    cout << texture << ' ';
                }
                cout << endl;
            }
            break;
        }
        case 2:
        {
            for (int i = 0; i < height; i++)
            {
                cout << "\t";
                for (int j = 0; j < width; j++)
                {
                    cout << ' ';
                    if ((i == 0 || i == height - 1) || (j == 0 || j == width - 1))
                        cout << texture;
                    else
                        cout << ' ';
                }
                cout << endl;
            }
            break;
        }
        default: { cout << endl << "В данном списке такой фигуры не имеется"; break; }
        }
        break;
    }
    case 4:
    {
        int a = 0;
        cout << "Введите высоту Треугольника: ";
        cin >> shape_size;

        cout << endl << endl;

        for (int i = shape_size; i > 0; i--)
        {
            cout << "\t";
            for (int j = 0; j < i; j++)
            {
                cout << ' ';
            }
            for (int i = 0; i <= a * 2; i++)
            {
                cout << texture;
            }
            cout << endl;
            a++;
        }
        break;
    }
    case 5:
    {

        break;
    }
    case 0: { break; }
    default: { cout << endl << "В данном списке такой фигуры не имеется"; break; }
    }

    cout << endl << endl << "Выход из программы! " << endl;
    system("pause");
    return 0;
}
