#include <iostream>
using namespace std;

int main()
{
    setlocale(0, "");
    cout << "\a\t\t***Приветствуем Вас в программе — \"Геометрические фигуры\"***\t\t\t\n";

    int figure, shape_size;
    cout << "\n\aВыберите геометрическую фигуру:" << endl;
    cout << "[1] Линия\n";
    cout << "[2] Квадрат\n";
    cout << "[3] Прямоугольник\n";
    cout << "[4] Треугольник\n";
    cout << "[0] Выход\n\n";

    cout << "Геометрическая фигура №: ";
    cin >> figure;

    switch (figure)
    {
    case 1:
    {
        int line_type;
        cout << "\n\aВыберите тип Линии:" << endl;
        cout << "[1] Линия\n";
        cout << "[2] Линия (Вертикальная)\n";
        cout << "[3] Слеш\n";
        cout << "[4] Слеш (Обратный)\n\n";

        cout << "Линия №: ";
        cin >> line_type;

        switch (line_type)
        {
        case 1:
        {
            cout << "Введите размер Линии: ";
            cin >> shape_size;
            cout << endl << endl << "\t";

            while (shape_size > 0)
            {
                cout << '*';
                shape_size -= 1;
            }
            break;
        }
        case 2:
        {
            cout << "Введите размер Линии (Вертикальной): ";
            cin >> shape_size;
            cout << endl << endl;

            while (shape_size > 0)
            {
                cout << "\t" << '*' << endl;
                shape_size -= 1;
            }
            break;
        }
        case 3:
        {
            cout << "Введите размер Слеша: ";
            cin >> shape_size;
            cout << endl;

            for (int i = shape_size; i > 0; i--)
            {
                cout << "\t";
                for (int j = 0; j <= shape_size; j++)
                {
                    if (i == j)
                    {
                        cout << '*';
                    }
                    else
                    {
                        cout << ' ';
                    }
                }
                cout << endl;
            }
            cout << endl;
            break;
        }
        case 4:
        {
            cout << "Введите размер Слеша (Обратного): ";
            cin >> shape_size;
            cout << endl;

            for (int i = 0; i < shape_size; i++)
            {
                cout << "\t";
                for (int j = 0; j <= shape_size; j++)
                {
                    if (i == j)
                    {
                        cout << '*';
                    }
                    else
                    {
                        cout << ' ';
                    }
                }
                cout << endl;
            }
            break;
        }
        default: { cout << endl << "В данном списке такой линии не имеется"; break; }
        }
        break;
    }
    case 2:
    {
        cout << "Введите размер Квадрата: ";
        cin >> shape_size;
        cout << endl << endl;

        for (int i = 0; i < shape_size; i++)
        {
            cout << "\t";
            for (int j = 0; j < shape_size; j++)
            {
                cout << '*' << ' ';
            }
            cout << endl;
        }
        break;
    }
    case 3:
    {
        int width, length;
        cout << "Введите длину Прямоугольника: ";
        cin >> length;
        cout << "Введите ширину Прямоугольника: ";
        cin >> width;

        cout << endl << endl;

        for (int i = 0; i < width; i++)
        {
            cout << "\t";
            for (int j = 0; j < length; j++)
            {
                cout << '*' << ' ';
            }
            cout << endl;
        }
        break;
    }
    case 4:
    {
        int a = 0;
        cout << "Введите высоту Треугольника: ";
        cin >> shape_size;

        cout << endl << endl;

        for (int i = shape_size; i > 0; i--)
        {
            cout << "\t";
            for (int j = 0; j < i; j++)
            {
                cout << ' ';
            }
            for (int i = 0; i <= a * 2; i++)
            {
                cout << '*';
            }
            cout << endl;
            a++;
        }
        break;
    }
    case 0: { break; }
    default: { cout << endl << "В данном списке такой фигуры не имеется"; break; }
    }

    cout << endl << endl << "Выход из программы! " << endl;
    system("pause");
    return 0;
}
